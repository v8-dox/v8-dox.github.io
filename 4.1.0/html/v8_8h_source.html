6209  V8_INLINE static bool IsValidSmi(intptr_t value) {
6210  // To be representable as an tagged small integer, the two
6211  // most-significant bits of 'value' must be either 00 or 11 due to
6212  // sign-extension. To check this we add 01 to the two
6213  // most-significant bits, and check if the most-significant bit is 0
6214  //
6215  // CAUTION: The original code below:
6216  // bool result = ((value + 0x40000000) & 0x80000000) == 0;
6217  // may lead to incorrect results according to the C language spec, and
6218  // in fact doesn't work correctly with gcc4.1.1 in some cases: The
6219  // compiler may produce undefined results in case of signed integer
6220  // overflow. The computation must be done w/ unsigned ints.
6221  return static_cast<uintptr_t>(value + 0x40000000U) < 0x80000000U;
6222  }
